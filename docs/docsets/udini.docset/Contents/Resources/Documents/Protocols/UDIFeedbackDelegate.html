UDIFeedbackDelegate

@protocol UDIFeedbackDelegate <NSObject>
@optional

/**
 * Tells the delegate that the bluetooth manager state has changed
 *
 * @param state Represents the current state of a CBManager.
 **/
- (void) bluetoothStateChanged:(CBManagerState)state;

/**
 * Tells the delegate a set of discovered devices have been discovered
 * These devices are pooled with a specific timeframe (see UDIOptions:deviceDiscoveryReportDelay)
 * This set of discovered devices will be unique.
 *
 * @param devices An array of devices.
 *
 * @see UDIOptions:deviceDiscoveryReportDelay
 *
 **/
- (void) didDiscoverDevices:(NSArray<UDIDevice *> * _Nonnull)devices;

/**
 * Tells the delegate a specific device has been discovered.
 * This delegate method will be fired for every detection, based on the advertisement time of the device
 *
 * @param device The discovered device.
 **/
- (void) didDiscoverDevice:(UDIDevice * _Nonnull)device;

/**
 * Tells the delegate a specific device has been lost.
 * If within a specific timeframe the device has not been discovered we assume it lost (see UDIOptions:deviceRangeTimeout)
 *
 * @param device The lost device.
 *
 * @see UDIOptions:deviceRangeTimeout
 *
 **/
- (void) didLoseDevice:(UDIDevice * _Nonnull)device;

/**
 * Tells the delegate a cooldown has been set for a specific device.
 * For every device action performed (successful or not) a defined cooldown will be set for the device.
 * This will ensure that automatic action are only performed with at
 * least a waiting period of the defined cooldown time (see UDIOptions:deviceCooldownInterval)
 * This will not affect manual action invocations.
 *
 * @param device The device associated with the cooldown.
 * @param time The cooldown time set for the device.
 *
 * @see UDIOptions:deviceCooldownInterval
 *
 **/
- (void) device:(UDIDevice * _Nonnull)device inCoolDownWithRemainingTime:(NSTimeInterval)time;

/**
 * Tells the delegate that a specific action did succeed
 *
 * @param device The device that describes the BLE device that has been written to.
 * @param deviceAction The deviceAction that has been written to, if nil it was a generic action (invokeOnProximity).
 */
- (void) device:(UDIDevice * _Nonnull)device didSucceedForAction:(UDIDeviceAction *)deviceAction;

/**
 * Tells the delegate that an error occured for a specific device
 *
 * @param device The device that describes the BLE device that has been written to.
 * @param deviceAction The deviceAction that has been written to, if nil it was a generic action (invokeOnProximity).
 * @param error The error that occured.
 */
- (void) device:(UDIDevice * _Nonnull)device forAction:(UDIDeviceAction *)deviceAction didFailWithError:(NSError *)error;

/**
 * Tells the delegate that the state has changed for a specific device
 *
 */
- (void) device:(UDIDevice * _Nonnull)device stateDidChange:(UDIDeviceState)state forAction:(UDIDeviceAction * _Nullable)action;

/**
 * Tells the delegate that the state has changed for a specific device
 *
 */
- (void) invokeOnProximityWillStartForDevice:(UDIDevice * _Nonnull)device withBadge:(UDIBadge * _Nonnull)badge;


@end
